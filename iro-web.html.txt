<script>
  // ── Sample seed data ──
  const seedIssues = [
    { id:'IRO-001', title:'Broken streetlight near Gate B', type:'AAUA Student',     category:'Infrastructure', urgency:'High',   status:'Resolved',    date:'2026-06-28', location:'Gate B, AAUA' },
    { id:'IRO-002', title:'Water scarcity in Akungba community', type:'Akungba Indigene', category:'Infrastructure', urgency:'High', status:'In Progress', date:'2026-07-01', location:'Akungba Market area' },
    { id:'IRO-003', title:'Conflict over land boundary in Iwaro', type:'Iwaro Indigene', category:'Social Conflict', urgency:'Med',  status:'Pending',     date:'2026-07-05', location:'Iwaro Road 3' },
    { id:'IRO-004', title:'Illegal dumping near Etioro school',   type:'Etioro Indigene', category:'Environmental',  urgency:'Med',  status:'In Progress', date:'2026-07-06', location:'Etioro Primary School' },
    { id:'IRO-005', title:'Exam timetable clash – 400 level',     type:'AAUA Student',    category:'Academic',       urgency:'Low',  status:'Resolved',    date:'2026-07-03', location:'Faculty of Science' },
  ];

  let issues = JSON.parse(localStorage.getItem('iro-issues') || 'null') || [...seedIssues];
  let urgencySelected = '';

  function saveIssues() { localStorage.setItem('iro-issues', JSON.stringify(issues)); }

  // ── Page navigation ──
  function showPage(id) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.querySelectorAll('.nav-links a').forEach(a => a.classList.remove('active'));
    document.getElementById('page-' + id).classList.add('active');
    document.getElementById('nav-' + id).classList.add('active');
    if (id === 'tracker') renderIssues(issues);
    window.scrollTo(0,0);
    return false;
  }

  // ── Urgency selection ──
  function selectUrgency(level) {
    urgencySelected = level;
    ['Low','Med','High'].forEach(l => {
      const el = document.getElementById('urg-' + l);
      el.classList.toggle('selected', l === level);
    });
  }

  // ── Submit report ──
  function submitReport() {
    const name     = document.getElementById('f-name').value.trim();
    const type     = document.getElementById('f-type').value;
    const category = document.getElementById('f-category').value;
    const title    = document.getElementById('f-title').value.trim();
    const desc     = document.getElementById('f-desc').value.trim();

    if (!name || !type || !category || !title || !desc || !urgencySelected) {
      alert('Please fill in all required fields and select an urgency level.');
      return;
    }

    const newIssue = {
      id:       'IRO-' + String(issues.length + 1).padStart(3,'0'),
      title,
      type,
      category,
      urgency:  urgencySelected,
      status:   'Pending',
      date:     new Date().toISOString().split('T')[0],
      location: document.getElementById('f-location').value.trim() || 'Not specified',
    };

    issues.unshift(newIssue);
    saveIssues();

    // Clear form
    ['f-name','f-email','f-title','f-desc','f-location'].forEach(id => document.getElementById(id).value = '');
    document.getElementById('f-type').value = '';
    document.getElementById('f-category').value = '';
    urgencySelected = '';
    ['Low','Med','High'].forEach(l => document.getElementById('urg-' + l).classList.remove('selected'));

    // Show toast
    const toast = document.getElementById('toast');
    toast.classList.add('show');
    setTimeout(() => toast.classList.remove('show'), 3500);

    setTimeout(() => showPage('tracker'), 1200);
  }

  // ── Render issues ──
  function renderIssues(list) {
    const container = document.getElementById('issues-list');
    if (!list.length) {
      container.innerHTML = `<div class="empty-state"><div class="empty-icon">📭</div><h3>No issues found</h3><p>No issues match this filter, or none have been submitted yet.</p></div>`;
      return;
    }
    container.innerHTML = list.map(issue => {
      const badgeClass = { Pending:'badge-pending', 'In Progress':'badge-progress', Resolved:'badge-resolved' }[issue.status] || '';
      const urgColor   = { Low:'#065f46', Med:'#92400e', High:'#991b1b' }[issue.urgency] || '#555';
      return `
        <div class="issue-card">
          <div class="issue-meta">
            <div class="issue-id">${issue.id} · ${issue.date}</div>
            <h4>${issue.title}</h4>
            <div class="issue-info">
              👤 ${issue.type} &nbsp;·&nbsp; 🗂 ${issue.category} &nbsp;·&nbsp; 📍 ${issue.location}
              &nbsp;·&nbsp; <span style="color:${urgColor};font-weight:600;">${issue.urgency} urgency</span>
            </div>
          </div>
          <div><span class="badge ${badgeClass}">${issue.status}</span></div>
        </div>`;
    }).join('');
  }

  // ── Filter ──
  function filterIssues(status, btn) {
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    const filtered = status === 'all' ? issues : issues.filter(i => i.status === status);
    renderIssues(filtered);
  }

  // Init tracker
  renderIssues(issues);
</script>
